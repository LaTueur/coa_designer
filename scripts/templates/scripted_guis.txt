increase_fixed_pattern = { 
	scope = culture
	effect = { 
		if = { 
			limit = { 
				NOT = { 
					exists = var:fixed_coa
				}
			}
			initialize_fixed_coa = yes
		}
		else_if = { 
			limit = { 
				var:fixed_pattern < 39
			}
			change_variable = { 
				name = fixed_pattern
				add = 1
			}
		}
		else = { 
			set_variable = { 
				name = fixed_pattern
				value = 0
			}
		}
	}
}
decrease_fixed_pattern = { 
	scope = culture

			}

		else = { 
			set_variable = { 
				name = fixed_pattern
				value = 39
			}
		}
	}
}
