initialize_fixed_coa = {
	scope = culture
	effect = {
		initialize_fixed_coa = yes
	}
}
increase_fixed_template = { 
	scope = culture
	effect = { 
		if = { 
			limit = { 
				var:fixed_template < 3
			}
			change_variable = { 
				name = fixed_template
				add = 1
			}
		}
		else = { 
			set_variable = { 
				name = fixed_template
				value = 0
			}
		}
	}
}
decrease_fixed_template = { 
	scope = culture
	effect = { 
		if = { 
			limit = { 
				var:fixed_template > 0
			}
			change_variable = { 
				name = fixed_template
				subtract = 1
			}
		}
		else = { 
			set_variable = { 
				name = fixed_template
				value = 3
			}
		}
	}
}
