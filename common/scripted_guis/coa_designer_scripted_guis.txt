change_fixed_template = { 
	scope = culture
	saved_scopes = { decrease}
	effect = { 
		if = { 
			limit = { 
				scope:decrease = yes
			}
			change_variable = { 
				name = fixed_template
				subtract = 1
			}
			if = { 
				limit = { 
					var:fixed_template < 0
				}
				set_variable = { 
					name = fixed_template
					value = max_fixed_template
				}
			}
		}
		else = { 
			change_variable = { 
				name = fixed_template
				add = 1
			}
			if = { 
				limit = { 
					var:fixed_template > max_fixed_template
				}
				set_variable = { 
					name = fixed_template
					value = 0
				}
			}
		}
	}
}
